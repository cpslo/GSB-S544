import requests
from bs4 import BeautifulSoup
import pandas as pd

def scrape_temps(month=None):
    base_url = "https://www.extremeweatherwatch.com/us-state-averages"

    # Construct URL based on the month provided
    if month is not None:
        my_url = base_url + "/month-" + month
    else:
        my_url = base_url

    # Fetch the page content
    response = requests.get(my_url)

    # Parse the page content
    soup = BeautifulSoup(response.content, "html.parser")

    # Find the first table in the page
    table = soup.find_all("table")[0]

    rows = []

    # Iterate over the table rows, skipping the header
    for row in table.find_all("tr")[1:]:
        cells = row.find_all('td')

        # Extract relevant data from the cells
        state = cells[0].text.strip()
        high = cells[1].text.strip()
        low = cells[2].text.strip()
        prep = cells[3].text.strip()

        # Append a dictionary of the row data
        rows.append({
            'state': state,
            'high': high,
            'low': low,
            'precipitation': prep
        })

    return pd.DataFrame(rows)

import requests
from bs4 import BeautifulSoup
import pandas as pd

def scrape_temps(month=None):
    base_url = "https://www.extremeweatherwatch.com/us-state-averages"

    # Construct URL based on the month provided
    if month is not None:
        my_url = base_url + "/month-" + month
    else:
        my_url = base_url

    # Fetch the page content
    response = requests.get(my_url)

    # Parse the page content
    soup = BeautifulSoup(response.content, "html.parser")

    # Find the first table in the page
    table = soup.find_all("table")[0]

    rows = []

    # Iterate over the table rows, skipping the header
    for row in table.find_all("tr")[1:]:
        cells = row.find_all('td')

        # Extract relevant data from the cells
        state = cells[0].text.strip()
        high = cells[1].text.strip()
        low = cells[2].text.strip()
        prep = cells[3].text.strip()

        # Append a dictionary of the row data
        rows.append({
            'state': state,
            'high': high,
            'low': low,
            'precipitation': prep
        })

    return pd.DataFrame(rows)
